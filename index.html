setup()

  • called once when the program starts
  • defines initial environment properties
            function setup(){
                noLoop();                // prevent infinite loop
                angleMode(DEGREES);      // set degrees instead of radians
                rectMode(CENTER);        // calculate rectangles from their centers
                pixelDensity(2);         // set resolution
                calculateValues();       // calculate all the sizes
                createCanvas(ww, hh);    // initialize an artboard
                person = new Person();   // initialize a character
                setGui();                // initialize the interface
            }

draw()

  • called directly after setup()
  • applies updates to the program
            function draw() {
                person.display();    // render a character
            }

Easy to scale

            let modulus = 12,                 // 12 basic module 
                microModule = modulus / 6,    // 2
                eyeWidth = microModule,       // 2
                eyeHeight = modulus / 2,      // 4
                earSize = modulus,            // 12
                fingerWidth = modulus * 2,    // 24
                headSize = modulus * 4,       // 48
                shoulderSize = modulus * 8,   // 96

random()

  • an extended equivalent of Math.random() in p5.js
  • picks a random value from a list or range
  • we apply random values to parameters of the character
  • and get unexpected and often amazing combinations
             this.makeMagic = () => {
                 this.hairStyle = random([ 'wavy', 'kichka', 'short' ]);
                 this.shoeStyle = random([ 'basic', 'boots', 'trendy' ]);
                 ...
                 this.leftHandAngle = random(minHandAngle, maxHandAngle);
                 this.rightHandAngle = random(minHandAngle, maxHandAngle);
                 ...
                 this.shoesTone = random(Object.values(clothesPalette));
                 this.skinTone = random(Object.values(skinPalette));
                 ...
             }
